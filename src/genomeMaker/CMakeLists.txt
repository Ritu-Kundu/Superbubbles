project(genome_maker)

#// Project flags

set(CMAKE_CXX_FLAGS " ${CMAKE_CXX_FLAGS} -std=c++14 -pthread")

        Reader.cpp
        Reader.h
        Creator.cpp
        Creator.h
        Randomiser.cpp

        GenomeMaker
